Python — Portal Login & Download
Auto-Download Report from Portal
from playwright.sync_api import sync_playwright import pandas as pd with sync_playwright() as p: browser = p.chromium.launch() page = browser.new_page() # Login to portal page.goto("https://yourportal.com/login") page.fill("#username", "your_user") page.fill("#password", "your_pass") page.click("#login-btn") # Navigate to report & download page.goto("https://yourportal.com/reports") with page.expect_download() as dl: page.click("#export-excel") dl.value.save_as("report.xlsx") browser.close()
Python — OCR Invoice Reader
Extract Data from Scanned Bill
import pytesseract, cv2 from PIL import Image import re def extract_invoice_data(image_path): img = cv2.imread(image_path) # Preprocess for better OCR gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY) thresh = cv2.threshold(gray, 150, 255, cv2.THRESH_BINARY)[1] text = pytesseract.image_to_string(thresh) # Extract key fields with regex amount = re.search(r'Total[:\s]+₹?([\d,]+)', text) date = re.search(r'\d{2}/\d{2}/\d{4}', text) gstin = re.search(r'[0-9]{2}[A-Z]{5}[0-9]{4}[A-Z]', text) return { 'amount': amount.group(1) if amount else '', 'date': date.group() if date else '', 'gstin': gstin.group() if gstin else '', }
